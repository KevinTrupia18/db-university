
-- 1. Studenti nati nel 1990
SELECT * FROM `students` WHERE YEAR(`date_of_birth`) = 1990;

-- 2. Corsi con più di 10 crediti (CFU)
SELECT * FROM `courses` WHERE `cfu` > 10;

-- 3. Studenti con più di 30 anni
SELECT * FROM `students` WHERE YEAR(CURDATE()) - YEAR(`date_of_birth`) > 30;

-- 4. Corsi del primo semestre del primo anno
SELECT * FROM `courses` WHERE `period` = 'I semestre' AND `year` = 1;

-- 5. Appelli del pomeriggio (dopo le 14) del 20/06/2020
SELECT * FROM `exams` WHERE `date` = '2020-06-20' AND `hour` > '14:00:00';

-- 6. Corsi di laurea magistrale
SELECT * FROM `degrees` WHERE `level` = 'magistrale';

-- 7. Numero dipartimenti
SELECT COUNT(*) FROM `departments`;

-- 8. Insegnanti senza numero di telefono
SELECT * FROM `teachers` WHERE `phone` IS NULL;



SOLUZIONI GROUP BY 


-- 1. Iscritti per ogni anno
SELECT YEAR(`enrolment_date`) as 'Anno', COUNT(*) as 'Totale' FROM `students` GROUP BY YEAR(`enrolment_date`);

-- 2. Insegnanti nello stesso ufficio/edificio
SELECT `office_address`, COUNT(*) FROM `teachers` GROUP BY `office_address`;

-- 3. Media voti di ogni appello
SELECT `exam_id`, AVG(`vote`) as 'Media Voto' FROM `exam_student` GROUP BY `exam_id`;

-- 4. Numero corsi di laurea per ogni dipartimento
SELECT `department_id`, COUNT(*) FROM `degrees` GROUP BY `department_id`;
